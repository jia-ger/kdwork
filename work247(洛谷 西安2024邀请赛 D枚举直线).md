#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin
#define int long long
#define unsigned long long ull
#define endl '\n'
typedef pair<int,int> PII;
const int N=2e5+10;
const int INF=1e18;
const int MOD=998244353;
//#define ll __int128


int n;
int a[200005],b[200005],ans,sum;
map<pair<int,int>,int> M;

void solve() {
  	cin>>n;
	for (int i=1;i<=n;i++)	cin>>a[i];
	for (int i=1;i<=n;i++){
		cin>>b[i];
		sum+=b[i];
	}
	for (int i=1;i<=n;i++){
		for (int j=0;j<=1000000/i;j++){
			int x=a[i]-(i-1)*j;
			if (x>=0){
				pair<int,int> p(x,j);
				M[p]+=b[i];
				ans=max(ans,M[p]);
			}
		}
	}
	cout<<sum-ans;
}



signed main() {

	cin.tie(0)->sync_with_stdio(0);

    
	int T=1;
//	cin>>T;

//   mt19937 rand(time(nullptr));
	while(T--)solve();

	return 0;
}
