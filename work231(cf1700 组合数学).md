> 分析：先看限制2和3 画出来环状图 发现是梯形 然后根据条件一自下而上进行组合数学计算

```c++
#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin
#define int long long
#define unsigned long long ull
#define endl '\n'
typedef pair<int,int> PII;
const int N=2e5+10;
const int INF=1e9;
const int MOD=1e9+7;
//#define ll __int128

const int p=998244353; 
int fac[N],inv[N];
int C(int n,int m){
	return fac[n]*inv[m]%p*inv[n-m]%p;
}

int qmi(long long a,int k){
	long long s=1;
	while(k){
		if(k&1)(s*=a)%=p;
		(a*=a)%=p;
		k>>=1;
	}
	return s;
}
void solve() {
	int n;
	cin>>n;
	vector<int>a(n+1);
	for(itn i=1;i<=n;i++)cin>>a[i];
	
	int cnt=0,ans=1;
	for(itn i=n;i>=1;i--){
		if(i<=n/2)cnt+=2;
		else if((n&1)&&(i==(n+1>>1)))cnt++;
		if(cnt<a[i]){
			cout<<0<<endl;return;
		}
		(ans*=C(cnt,a[i]))%=p;
		cnt-=a[i];
	}
	if(cnt){
		cout<<0<<endl;return;
	}
	
	cout<<ans<<endl;
}

signed main() {
    
    cin.tie(0)->sync_with_stdio(0);

	fac[0]=inv[0]=1;
	for(int i=1;i<N;i++){
		fac[i]=fac[i-1]*i%p;
		inv[i]=qmi(fac[i],p-2);
	}
	
	int T=1;
	cin>>T;
//   mt19937 rand(time(nullptr));
	while(T--)solve();

	return 0;
}
```
