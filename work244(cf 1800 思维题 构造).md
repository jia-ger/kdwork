```c++
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define endl '\n'

void solve() {
    int n, k;
    cin>>n>>k;
    vector<int> a(n), b(n);
    for (int i = 0; i < n; ++i) cin >> a[i];
    for (int i = 0; i < n; ++i) cin >> b[i];

    vector<int> v(k, -1);
    
    for (int j = 0; j < k; ++j) {
        bool all_a_same = true;
        int first_a = a[j];
        for (int i = j + k; i < n; i += k) {
            if (a[i] != first_a) {
                all_a_same = false;
                break;
            }
        }

        if (!all_a_same) {
            // 如果 a 组内值不全相同，则 b[i] 必须等于 a[i]
            for (int i = j; i < n; i += k) {
                if (b[i] != -1 && b[i] != a[i]) {
                    cout << "NO" << endl;
                    return;
                }
            }
            v[j] = a[j]; // 此时 b[j] 被强制固定为 a[j]
        } else {
            // 如果 a 组内值全相同，则 b[i] 必须全部等于同一个值
            int fixed_v = -1;
            for (int i = j; i < n; i += k) {
                if (b[i] != -1) {
                    if (fixed_v == -1) fixed_v = b[i];
                    else if (fixed_v != b[i]) {
                        cout << "NO" << endl;
                        return;
                    }
                }
            }
            v[j] = fixed_v; // 可能是 -1 (表示灵活)
        }
    }

    // 多重集检查：{v_j | v_j != -1} 必须是 {a[0...k-1]} 的子多重集
    vector<int> count_u(n + 1, 0);
    for (int j = 0; j < k; ++j) {
        count_u[a[j]]++;
    }

    for (int j = 0; j < k; ++j) {
        if (v[j] != -1) {
            if (count_u[v[j]] <= 0) {
                cout << "NO" << endl;
                return;
            }
            count_u[v[j]]--;
        }
    }

    cout << "YES" << endl;
}

signed main() {
    cin.tie(0)->sync_with_stdio(0);

	int T=1;
	cin>>T;

//   mt19937 rand(time(nullptr));
	while(T--)solve();
	
    return 0;
}
```
