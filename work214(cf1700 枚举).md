```c++
#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin
#define int long long
#define endl '\n'
typedef pair<int,int> PII;
const int N=2e5+10;
const int INF=1e18;
const int MOD=998244353;
#define ll __int128

//发现这道题原本的黑色块 覆盖情况只能是两种 要么被2*2的方块覆盖 要么被两条斜线覆盖 
char g[105][105];
int n; 
void solve()
{
  cin>>n;
  int cnt=0;
  for(itn i=1;i<=n;i++)
  for(itn j=1;j<=n;j++)
  {
  	cin>>g[i][j];
  	if(g[i][j]=='#')cnt++;
  }
  if(cnt==1||n==1)
  {
  	cout<<"YES"<<endl;return;
  }
  for(itn i=2;i<=n;i++)//2*2覆盖 
   for(itn j=2;j<=n;j++)
   {
     int s=(g[i][j]=='#')+(g[i-1][j-1]=='#')+(g[i-1][j]=='#')+(g[i][j-1]=='#');
	 if(s==cnt)
	 {
	   cout<<"YES"<<endl;return;	
	 }	
   }
  for(int i=1-n;i<=n-2;i++)//枚举右下斜线的差 右下有规律x+1 y也+1 所以横纵坐标差为定值 
  {
  	int s=0;
  	for(int j=1;j<=n;j++){if(i+j<1||i+j>n)continue;s+=g[j][i+j]=='#';}
  	for(int j=1;j<=n;j++){if(i+j+1<1||i+j+1>n)continue;s+=g[j][i+j+1]=='#';}//下面那一层 
  	if(cnt==s)
  	{
  	 cout<<"YES"<<endl;return;	
	}
  }      
  for(int i=2;i<=n*2-1;i++)//枚举左下斜线的和 左下横纵坐标和为定值 
  {
    int s=0;
    for(int j=1;j<=n;j++){if(i-j<1||i-j>n)continue;s+=g[j][i-j]=='#';}
    for(int j=1;j<=n;j++){if(i-j+1<1||i-j+1>n)continue;s+=g[j][i-j+1]=='#';}
    if(cnt==s){cout<<"YES"<<endl;return ;} 	
  } 
  cout<<"NO"<<endl; 
}



signed main(){
   cin.tie(0)->sync_with_stdio(0);	 
   int T=1; 

   cin>>T;
   
  
   while(T--)solve();
   
    return 0;
}
```
