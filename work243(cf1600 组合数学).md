> 分析：当前缀和在某两位上的值相同时 sumx=sumy 则有 x+1,y 的1的数量和0的数量一致 贡献为 (x+1)*(n-y+1)

```c++
#define int long long
const int MOD = 1e9+7;
int t,n;
string s;

void solve(){
	int res = 0;
	cin >> s;
	n = s.size(); s = " " + s;

	map<int,int> mp;
	mp[0] = 1;
	int sum = 0;
	for(int i=1;i<=n;i++){
		if(s[i] == '1') sum++;
		else sum--;
		res = (res+mp[sum]*(n-i+1))%MOD;
		mp[sum] = (mp[sum] + i+1)%MOD;
	}
	cout << (res+MOD)%MOD << '\n';
}
```
