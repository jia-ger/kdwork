```c++
#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin
#define int long long
#define unsigned long long ull
#define endl '\n'
typedef pair<int,int> PII;
const int N=15e6+10;
const int INF=1e18;
const int MOD=998244353;
//#define ll __int128



int m[N];//26^0+26^1+26^2+26^3+26^4<1.5e6 所以可以开数组存压缩后的答案
vector<int> S[6];//最多有5位匹配

//返回子串被压缩后的编号
int id(int st,string s){//st表示子串的匹配情况
	for(int i=s.length();i<5;i++){
		if(st>>i&1)	return -1;//超过s本身长度还匹配成功肯定 是非法的子串
	}
	int res = 0,b = 1;
	for(int i=0;i<s.length();i++){
		if(st>>i&1)	res+=b*(s[i]-'a'+1);//此位是匹配的 
		b*=27;
	}
	return res;
}

void solve() {
    itn q;
    cin>>q;
    while(q--){
    	int op;
    	string s;
    	cin>>op>>s;
    	if(op==1){
    		int x;
    		cin>>x;
    		for(int i=0;i<=s.length();i++){
    			for(int st:S[i]){//枚举s对应的所有 子串的匹配情况
    				int now=id(st,s);
    				if(now==-1)continue;
    				if(!m[now]||m[now]>x)m[now]=x;
				}
			}
		}
		else{
			for(int l=s.length();l>=0;l--){//匹配的可能性从多到少排列 
				int res = INF;
				for(int st:S[l]){
					int now = id(st,s);
					if(now==-1)	continue;
					if(m[now])//存在一个这样的匹配序列 有对应编号
					res = min(res,m[now]);
				}
				if(res<INF){
					cout<<res<<endl;break;
				}
			}
		}
	}
}



signed main() {

	cin.tie(0)->sync_with_stdio(0);
    for(int i=0;i<(1<<5);i++)S[__builtin_popcount(i)].push_back(i);//不同的二进制对应有几位匹配的
	int T=1;
//	cin>>T;

//   mt19937 rand(time(nullptr));
	while(T--)solve();

	return 0;
}
```
