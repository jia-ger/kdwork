```c++
#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin
#define int long long
#define endl '\n'
typedef pair<int,int> PII;
const int N=2e5+10;
const int INF=1e7;
const int MOD=998244353;
//#define ll __int128

int n;
int a[2][N];
int lmax[N],lmin[N],rmax[N],rmin[N];//预处理上下两行前后缀 
int val[N*2];//1-2*n 范围不要开小了！ 
void solve()
{
 	
  
  cin>>n;
  lmin[0]=INF,rmin[n+1]=INF,rmax[n+1]=-INF,lmax[0]=-INF;//防止最小值读取到0或最大值读取到上次的脏数据 
  for(int i=1;i<=n;i++)cin>>a[0][i],lmax[i]=max(lmax[i-1],a[0][i]),lmin[i]=min(lmin[i-1],a[0][i]);
  for(itn i=1;i<=n;i++)cin>>a[1][i];
  for(int i=n;i>=1;i--)rmax[i]=max(rmax[i+1],a[1][i]),rmin[i]=min(rmin[i+1],a[1][i]);
     
   int ans=0; 

   for(int i=1;i<=2*n;i++)val[i]=2*n+1;
   for(int i=1,L,R;i<=n;i++)
   {
     L=min(lmin[i],rmin[i]);
     R=max(lmax[i],rmax[i]);
     val[L]=min(val[L],R);
   }
    for(int i=2*n,cur=2*n+1;i>=1;i--)//直接枚举L 逆序可以保证取到的永远是最近的r 
	{
     cur=min(cur,val[i]);
     ans+=2*n-cur+1; 
    }
  
  cout<<ans<<endl;
}



signed main(){
   cin.tie(0)->sync_with_stdio(0);	 
   int T=1; 

   cin>>T;
   
  
   while(T--)solve();
   
    return 0;
}
```
