#include <bits/stdc++.h>

int t,n,sum,num[200005];
set<int> so;
map<int,int> mo; 
void solve()
{
    cin>>n;sum=0;
	int awa=0;
	so.clear();
	for(int i=1;i<=n;i++) cin>>num[i],sum+=num[i];//可以发现每次循环右移一位 以及第二次循环会移除只有一项的数 
	for(int i=1;i<=n;i++){
		if(so.find(num[i])!=so.end()) awa=max(awa,num[i]);
		so.insert(num[i]);num[i]=awa;sum+=awa;
	}
	so.clear();awa=0;
	for(int i=1;i<=n;i++){
		if(so.find(num[i])!=so.end()) awa=max(awa,num[i]);
		so.insert(num[i]);num[i]=awa;sum+=awa;
	}//暴力循环两次 
	for(int i=1;i<=n;i++) sum+=(n-i)*num[i];
	cout<<sum<<endl;
}



signed main(){
   cin.tie(0)->sync_with_stdio(0);	 
   int T=1; 

   cin>>T;
   
  
   while(T--)solve();
   
    return 0;
}
