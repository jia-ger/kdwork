https://codeforces.com/contest/1907/problem/E

```c++
#include<bits/stdc++.h>
using namespace std;
int t,x[10],n;
int main(){
	for(int i=0;i<=9;i++)
		for(int j=0;j<=9;j++)
			for(int k=0;k<=9;k++)
				if(i+j+k<10) x[i+j+k]++;//存储数 肯定不会进位，进位的话每一位之和就比原来大9 
	cin>>t;//多组数据 
	while(t--)
	{
		cin>>n;
		long long sum=1;//十年oi一场空，不开longlong见祖宗 
		while(n)
		{
			sum*=x[n%10];//求出数 
			n/=10;
		}
		cout<<sum<<endl;//多组数据必要输出换行 
	}
	return 0;
}
```
