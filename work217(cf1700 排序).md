#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin
#define int long long
#define endl '\n'
typedef pair<int,int> PII;
const int N=2e5+10;
const int INF=1e7;
const int MOD=998244353;
//#define ll __int128

int a[200005], b[200005];
int n, m;
int get_max(int l1, int r1, int l2, int r2){//统计区间的最大保留数量
	if(l1 > r1 || l2 > r2) return 0;
	int ct1 = 0, ct2 = 0;
	for(int i = l1; i <= r1; i++){//记录 A 中 [l1,r1] 的 m 的数量
		if(a[i] == b[m]) ct1++;
	}
	for(int i = l2; i <= r2; i++){//记录 A 中 [l2,r2] 的 m 的数量
		if(a[i] == b[m]) ct2++;
	}
	if(l1 == l2) return ct1;//注意特判最中间的区间，只能加一次
	else return min(ct1, ct2) * 2;//最小值的两倍就是最大保留数量
}
void solve()
{
  	cin >> n >> m;
	for(int i = 1; i <= n; i++) cin >> a[i], b[i] = a[i];
	
	nth_element(b + 1, b + m, b + n + 1);//求第 k 小的数
    //调用这个函数后 左边都<=b[m] 右边都>=b[m] 
    
	vector <int> v;//用 vector 存储小于 m 的数的下标
	for(int i = 1; i <= n; i++){
		if(a[i] < b[m]) v.push_back(i);
	}
	if(v.size() == 0){//如果 v 为空，说明留下的数全为 m，显然回文
		cout << "YES\n";
		return;
	}
    
    //判断现在的数组是否回文
	vector <int> v2;
	for(int i = 0; i < v.size(); i++) v2.push_back(a[v[i]]);//由于存储的是下标，所以需要求出 A 数组中它的值
	vector <int> v3 = v2;
	reverse(v3.begin(), v3.end());//反转判回文
	if(v3 != v2){//如果所有小于 m 的数都不能构成回文数组，则原数组无法变成回文数组
		cout << "NO\n";
		return; 
	}
    
    //统计贡献
	int add = m - v.size() - 1;//至少得留下 k-siz-1 个 m
	int tmp = get_max(1, v.front() - 1, v.back() + 1, n);
	for(int i = 0; i * 2 + 1 < v.size(); i++){//构成回文串 
		tmp += get_max(v[i] + 1, v[i + 1] - 1, v[v.size() - i - 2] + 1, v[v.size() - i - 1] - 1);//累加区间贡献
	}
	if(tmp >= add) cout << "YES\n";
	else cout << "NO\n";
}



signed main(){
   cin.tie(0)->sync_with_stdio(0);	 
   int T=1; 

   cin>>T;
   
  
   while(T--)solve();
   
    return 0;
}
