```c++
#include<bits/stdc++.h>
using namespace std;

long long r,g,b;

struct node{
	int num;
	char c;
}a[5];

bool cmp(node x,node y){
	return x.num > y.num;
}

int main(){
	int t;
	cin >> t;
	while(t--){
		cin >> r >> g >> b;
		a[0].c='R';
		a[0].num=r;
		a[1].c='G';
		a[1].num=g;
		a[2].c='B';
		a[2].num=b;
		sort(a,a+3,cmp);
		while(a[0].num < a[1].num + a[2].num){
			if(a[1].num>0 && a[2].num > 0){
				cout << a[2].c << a[1].c;
				a[1].num--;
				a[2].num--;
			}
		}
		while(a[1].num){
			cout << a[0].c << a[1].c;
			a[0].num--;
			a[1].num--;
		}
		while(a[2].num){
			cout << a[0].c << a[2].c;
			a[0].num--;
			a[2].num--;
		}
		if(a[0].num)
			cout << a[0].c;
		cout << endl;
	}
	return 0;
}
```
