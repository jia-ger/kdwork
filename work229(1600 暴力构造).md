#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin

void solve() {
  itn n,m;
  cin>>n>>m;
  itn s1=0;
   
  vector<vector<int>>a(n+1);
  vector<int>b(m+1,0);
  
  for(itn i=1;i<=n;i++){
  	a[i].resize(m+1);
  	for(itn j=1;j<=m;j++)
  	{
	cin>>a[i][j];
  	if(a[i][j]==1)s1++,b[j]++;
    }
  } 
  
  itn x1=s1/2;
  int x2=(s1+1)/2;
  int ans=x1*x2;
  
  cout<<ans<<endl; 
  
  for(int i=1;i<=m;i++)b[i]+=b[i-1];
  
  auto it=upper_bound(b.begin(),b.end(),x1);
  int index=distance(b.begin(),it)-1;
  if(b[index]==x1){
  	for(int i=1;i<=index;i++)cout<<'R';
  	for(int i=1;i<=n;i++)cout<<'D';
  	for(itn i=index+1;i<=m;i++)cout<<'R';
  	cout<<endl;
  } 
  else{
  	int res=x1-b[index];
  	int p=-1;
  	for(int i=n;i>=1;i--){
  		if(a[i][index+1]==1)res--;
  		if(res==0){
  			p=i;break;
		  }
	  }
//	cout<<"p="<<p<<endl;
//	cout<<"index="<<index<<endl;
	
	for(int i=1;i<=index;i++)cout<<'R';
	for(itn i=1;i<p;i++)cout<<'D';
	cout<<'R';
	for(int i=p;i<=n;i++)cout<<'D';
	for(int i=index+2;i<=m;i++)cout<<'R';
	cout<<endl;  
  }
}

signed main() {
    
    cin.tie(0)->sync_with_stdio(0);

	int T=1;
	cin>>T;
//   mt19937 rand(time(nullptr));
	while(T--)solve();

}
