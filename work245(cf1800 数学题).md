> 分析 ：首先如果出现1不是前导的情况则直接-1无解 $$a_i->a_i^2$$ 执行 $$k_i$$ 次 $$a_i->a_i^{2^{k_i}}$$  所以我们要让
>  $$a_i^{2^{k_i}}$$ $$\geq$$  $$a_{i-1}^{2^{k_{i-1}}$$ 两边同时log运算最后得出 $$k_i$$ = $$k_{i-1}$$ + $$log2{\frac{lga_{i-1}}{lga_i}}$$

```c++
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define endl '\n'
const double eps = 1e-9;
int n,lst,sum,use,flag,x;
void solve() {
    	cin >> n >> lst, sum = use = 0, flag = 0;
		for (int i = 1; i < n; ++ i) {
			cin >> x;
			if (lst != 1 && x == 1)
				flag = 1;
			if (flag || x == 1)
				continue ;
			//use表示这个数操作几次
			//-1.0*use 意味着这个数不需要操作 use为0 	
			use += max (-1.0 * use, ceil (log2 (log2 (lst) / log2 (x)) - eps));
			sum += use;
			lst = x;
		}
		cout << (flag ? -1 : sum) << endl;
}

signed main() {
    cin.tie(0)->sync_with_stdio(0);

	int T=1;
	cin>>T;

//   mt19937 rand(time(nullptr));
	while(T--)solve();
	
    return 0;
}
```
