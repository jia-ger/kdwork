```c++
#include <bits/stdc++.h>
using namespace std;
#define itn int
#define icn cin
#define int long long
#define unsigned long long ull
#define endl '\n'
typedef pair<int,int> PII;
const int N=2e5+10;
const int INF=1e18;
const int MOD=998244353;
//#define ll __int128


int n,q,it[N],a[N];
void solve() {
	int l,r,k,cnt,sum,cntt,summ;
	cin>>n>>q;
	for(itn i=1;i<=n;i++){
		cin>>a[i];
		it[a[i]]=i;
	}
	while(q--)
		{
			cin>>l>>r>>k;
			cnt = sum = cntt = summ = 0;
			if(it[k]<l||it[k]>r)
			{
				cout<<-1<<" ";
				continue;
			}
			while(l<=r)
			{
				int mid = (l+r)>>1;
				if(a[mid]==k)
				{
					break;
				}
				if(mid<it[k])// k出现在区间右侧
				{
					l = mid + 1;
					if(a[mid]>k)// 这个数需要变成小于 k的数
					{
						cntt++;
					}
					cnt++;
				}
				else
				{
					r = mid - 1;
					if(a[mid]<k)//这个数需要变成大于k的数
					{
						summ++;
					}
					sum++;
				}
			}
			if(cnt>k-1||sum>n-k)//如果小于他的数或大于他的数不符合数量要求则输出-1
			{
				cout<<-1<<" ";
			}
			else
			{
				cout<<max(summ,cntt)*2<<" ";
			}
		}
		cout<<endl;
}



signed main() {

	cin.tie(0)->sync_with_stdio(0);
	
	int T=1;
	cin>>T;

//   mt19937 rand(time(nullptr));
	while(T--)solve();

	return 0;
}
```
