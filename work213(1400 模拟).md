void solve()
{
       int maxa=0,n,k;

        for (int i=1;i<=n;i++){
            cin>>a[i];
            maxa=max(maxa,a[i]);  //计算最大值
        }
        int l=maxa,r=maxa+k-1;  //全部灯都亮起的时刻一定在l和r中间
        for (int i=1;i<=n;i++){
            if (((maxa-a[i])/k)%2==1) l=max(l,a[i]+((maxa-a[i])/k+1)*k);  //奇数次转化就是灭的

        }
        if (l<=r) cout<<l<<'\n';  //如果存在这样的时刻，输出最小时刻
        else cout<<-1<<'\n';  //不存在这样的时刻，输出-1
  
}



signed main(){
   cin.tie(0)->sync_with_stdio(0);	 
   int T=1; 

   cin>>T;
   
  
   while(T--)solve();

}
