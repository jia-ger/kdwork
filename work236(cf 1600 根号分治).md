//根号分治 a[i]*a[j]=j-i<n 则 min(a[i],a[j])<根号n  分情况讨论大于根号n和小于根号n
	
void solve() {
	int n,s;
    cin>>n;
    vector<int> a(n+5);
    for(int i=1;i<=n;i++)cin>>a[i];
    s=sqrt(n);
    int ans=0;
    for(int i=1;i<=s;i++){        //a[i]<=s 枚举a[i] 1-n枚举j
        for(int j=1;j<=n;j++){
            int idx=j-a[j]*i;
            if(idx<=0)continue;
            ans+=(idx<j&&a[idx]==i);
        }
    }
    for(int i=1;i<=n;i++){       //a[i]>s 枚举可能的a[j] 
        if(a[i]<=s)continue;
        for(int j=1;j*a[i]<=n;j++){
            int idx=i+a[i]*j;
            ans+=(idx<=n&&a[idx]==j);
        }
    }
    cout<<ans<<endl;
}



signed main() {

	cin.tie(0)->sync_with_stdio(0);
	
	int T=1;
	cin>>T;

//   mt19937 rand(time(nullptr));
	while(T--)solve();

	return 0;
}
